\section{Mathematical Formulation}

\begin{frame}{Integer Linear Programming Model}
\begin{block}{Decision Variables}
\[x_{s,sh} \in \{0,1\}\]
Binary variable: 1 if staff $s$ is assigned to shift $sh$, 0 otherwise
\end{block}

\begin{alertblock}{Objective Function (Minimize Cost)}
\[\text{Minimize } Z = \sum_{s \in S} \sum_{sh \in Sh} x_{s,sh} \cdot c_s \cdot d_{sh}\]
Where:
\begin{itemize}
\item $c_s$ = cost per hour for staff $s$ (\$/hr)
\item $d_{sh}$ = duration of shift $sh$ (hours)
\item $S$ = set of all staff members
\item $Sh$ = set of all shifts
\end{itemize}
\end{alertblock}
\end{frame}

\begin{frame}{Constraints}
\begin{block}{1. Shift Coverage Requirement}
\[\sum_{s \in S} x_{s,sh} \geq r_{sh} \quad \forall sh \in Sh\]
Each shift must have at least $r_{sh}$ staff assigned
\end{block}

\begin{block}{2. Maximum Working Hours}
\[\sum_{sh \in Sh} x_{s,sh} \cdot d_{sh} \leq h_{max,s} \quad \forall s \in S\]
Total hours worked by staff $s$ cannot exceed their maximum $h_{max,s}$
\end{block}

\begin{block}{3. Availability Constraint}
\[x_{s,sh} = 0 \text{ if } sh \notin A_s\]
Staff can only be assigned to shifts they are available for ($A_s$ = availability set)
\end{block}
\end{frame}

\begin{frame}{Additional Constraints}
\begin{block}{4. One Shift Per Day Constraint}
For each day $d$ and staff $s$:
\[x_{s,d\_AM} + x_{s,d\_PM} \leq 1\]
Staff cannot work both morning and afternoon shift on the same day
\end{block}

\begin{exampleblock}{Solution Method}
\begin{itemize}
\item Uses PuLP (Python Linear Programming library)
\item CBC (Coin-or Branch and Cut) solver
\item Binary Integer Programming
\item Returns optimal assignment or infeasibility status
\end{itemize}
\end{exampleblock}
\end{frame}

\section{Patient Flow Simulation}

\begin{frame}{Queue Theory Integration}
\begin{block}{M/M/c Queueing Model}
System modeled as multi-server queue:
\begin{itemize}
\item \textbf{M}: Markovian (exponential) arrivals - patient arrivals follow Poisson process
\item \textbf{M}: Markovian (exponential) service times
\item \textbf{c}: Number of servers (healthcare staff)
\end{itemize}
\end{block}

\begin{alertblock}{Traffic Intensity (Bottleneck Detection)}
\[\rho = \frac{\lambda}{\mu \cdot c}\]
Where:
\begin{itemize}
\item $\lambda$ = arrival rate (patients/hour)
\item $\mu$ = service rate (patients/hour per server)
\item $c$ = number of servers
\item $\rho < 1.0$ required for system stability
\end{itemize}
\end{alertblock}
\end{frame}

\begin{frame}{Bottleneck Classification}
\begin{block}{Four-Level Detection System}
\begin{itemize}
\item \textbf{Critical} ($\rho \geq 0.95$): System at breaking point
\item \textbf{Warning} ($0.85 \leq \rho < 0.95$): High utilization, delays likely
\item \textbf{Caution} ($0.75 \leq \rho < 0.85$): Monitor closely
\item \textbf{Healthy} ($\rho < 0.75$): Adequate capacity
\end{itemize}
\end{block}

\begin{exampleblock}{Little's Law Verification}
\[L = \lambda W\]
\begin{itemize}
\item $L$ = average number in system
\item $\lambda$ = arrival rate
\item $W$ = average time in system
\item Used to validate simulation results
\end{itemize}
\end{exampleblock}
\end{frame}

def analyze_infeasibility(cfg):
    issues = []
    suggestions = []
    
    # Check 1: Insufficient staff for shifts
    for shift, required in shift_requirements.items():
        available = count_available_staff(shift)
        if available < required:
            issues.append(f"{shift}: needs {required}, 
                          only {available} available")
    
    # Check 2: Overworked staff
    for staff in critical_staff:
        hours_needed = calculate_critical_hours(staff)
        if hours_needed > max_hours[staff]:
            suggestions.append(f"Increase {staff} max_hours 
                               from {max_hours} to {hours_needed}")
    
    return {"has_issues": True, "issues": issues, 
            "suggestions": suggestions}

function collectConfigFromForm() {
    const staffConfig = {};
    const container = document.getElementById('staff-config-container');
    
    Array.from(container.children).forEach((card, index) => {
        const name = card.querySelector('[data-field="name"]').value;
        const cost = parseFloat(card.querySelector('[data-field="cost"]').value);
        const maxHours = parseInt(card.querySelector('[data-field="max_hours"]').value);
        
        // Get availability from checkboxes
        const availability = [];
        card.querySelectorAll('input[type="checkbox"][data-shift]').forEach(cb => {
            if (cb.checked) availability.push(cb.dataset.shift);
        });
        
        staffConfig[name] = {cost, max_hours: maxHours, availability};
    });
    return {optimiser: {staff: staffConfig, ...}};
}

\item \textbf{Backend}:
\begin{itemize}

\item Reloads config from file (get\_current\_config)
\item Extracts staff, costs, hours, availability
\item Creates PuLP LP model
\item Adds decision variables (binary)
\item Sets objective function (minimize cost)
\item Adds constraints (coverage, hours, availability, one-shift-per-day)
\end{itemize}

\begin{frame}{Optimization Workflow (continued)}
\begin{enumerate}
\setcounter{enumi}{3}

\begin{itemize}

\item Gets status (Optimal, Infeasible, Unbounded)
\item If Optimal: extract assignments, calculate cost
\item If Infeasible: call analyze\_infeasibility()

\item \textbf{Infeasibility Analysis}:
\begin{itemize}
\item Check each shift for sufficient staff

\item Generate specific suggestions

\item \textbf{Response}:
\begin{itemize}
\item Return JSON with status, cost, assignments OR
\item Return JSON with status="infeasible", issues, suggestions
\end{itemize}
\item \textbf{Frontend}: Display results or infeasibility analysis
\end{enumerate}

\begin{frame}{Example Infeasibility}
\begin{alertblock}{Test Case: Nurse\_A Wed-Fri Only}
\textbf{Configuration Change:}
\begin{itemize}
\item Nurse\_A availability: ONLY Wed\_AM, Thu\_AM, Fri\_AM (removed Mon, Tue)
\item Tech\_D max\_hours: 40 (reduced from 48)
\end{itemize}

\vspace{0.3cm}

\textbf{System Response:}
\begin{itemize}
\item \textbf{Status}: Infeasible
\item \textbf{Issues}:
\begin{itemize}
\item Tech\_D: Needed for 7 critical shifts (56hrs) but max\_hours is 40
\item Mon\_AM: Exactly 2 staff available - no flexibility
\end{itemize}
\item \textbf{Suggestions}:
\begin{itemize}
\item Increase Tech\_D's max\_hours from 40 to 56 or more
\item Add backup staff for Monday morning
\end{itemize}
\end{itemize}
\end{alertblock}
