\section{Problem Statement}

\begin{frame}{Healthcare Staffing Challenge}
    \begin{block}{The Problem}
        Healthcare facilities face critical challenges in staff scheduling:
        \begin{itemize}
            \item \textbf{Patient demand} varies throughout the week
            \item \textbf{Staff availability} is constrained
            \item \textbf{Cost optimization} while maintaining service quality
            \item \textbf{Bottleneck identification} in patient flow
        \end{itemize}
    \end{block}
    
    \begin{exampleblock}{Solution Approach}
        Integrated web platform combining:
        \begin{itemize}
            \item Integer Linear Programming for staff optimization
            \item M/M/c Queue Theory for bottleneck analysis
            \item Modern web interface for real-time configuration
        \end{itemize}
    \end{exampleblock}
\end{frame}

\section{Mathematical Model}

\begin{frame}{Integer Linear Programming Formulation}
    \begin{block}{Decision Variables}
        $x_{is} \in \{0,1\}$ : Staff member $i$ works shift $s$
    \end{block}
    
    \begin{block}{Objective Function}
        \[
        \text{Minimize: } \sum_{i \in \text{Staff}} \sum_{s \in \text{Shifts}} c_i \cdot x_{is}
        \]
        where $c_i$ is the cost per shift for staff member $i$
    \end{block}
    
    \begin{block}{Key Constraints}
        \begin{itemize}
            \item Shift requirements: $\sum_{i} x_{is} \geq R_s$ for each shift $s$
            \item Max hours: $\sum_{s} x_{is} \leq H_i$ for each staff $i$
            \item Availability: $x_{is} = 0$ if staff $i$ unavailable for shift $s$
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Queue Theory Integration}
    \begin{block}{M/M/c Queue Model}
        \begin{itemize}
            \item $\lambda$ : Patient arrival rate (patients/hour)
            \item $\mu$ : Service rate per staff (patients/hour)
            \item $c$ : Number of servers (staff members)
            \item $\rho = \frac{\lambda}{\mu \cdot c}$ : Traffic intensity
        \end{itemize}
    \end{block}
    
    \begin{alertblock}{Bottleneck Detection Criteria}
        \begin{itemize}
            \item \textcolor{red}{Critical}: $\rho \geq 0.95$ (System overload)
            \item \textcolor{orange}{Warning}: $0.85 \leq \rho < 0.95$ (High congestion)
            \item \textcolor{yellow}{Caution}: $0.75 \leq \rho < 0.85$ (Moderate load)
            \item \textcolor{green}{Healthy}: $\rho < 0.75$ (Normal operation)
        \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Dynamic Configuration Reload}
    \begin{alertblock}{Problem}
        Configuration changes weren't reflected in optimization results

    \begin{exampleblock}{Solution}

    \begin{lstlisting}[language=Python, basicstyle=\ttfamily\tiny]
def get_current_config():
    """Reload config from file to get latest changes"""
    with open('config.json', 'r') as f:
        return json.load(f)

def run_optimisation():
    config = get_current_config()  # Fresh data every time
    # ... optimization logic ...
    \end{lstlisting}
    
    \begin{block}{Impact}
        Users can modify configurations and see immediate results
    \end{block}
\end{frame}

\begin{frame}[fragile]{Infeasibility Analysis Engine}
    \begin{lstlisting}[language=Python, basicstyle=\ttfamily\tiny]
def analyze_infeasibility(config):
    issues = []
    suggestions = []
    
    # Check 1: Insufficient staff
    available_staff = len(config['staff'])
    max_requirement = max(config['shift_requirements'].values())
    if available_staff < max_requirement:
        issues.append("Insufficient staff available")
        suggestions.append(f"Hire {max_requirement - available_staff} more staff")
    
    # Check 2: No flexibility (staff can't cover shifts)
    for shift in config['shift_requirements']:
        available_for_shift = [s for s, data in config['staff'].items() 
                               if shift in data['availability']]
        if len(available_for_shift) < config['shift_requirements'][shift]:
            issues.append(f"Not enough staff available for {shift}")
    
    # ... more checks ...
    return {"issues": issues, "suggestions": suggestions}
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{REST API Endpoints}
    \begin{block}{Core API Routes}
        \begin{description}
            \item[POST /api/simulate] Run patient flow simulation
            \item[POST /api/optimize] Execute staff optimization
            \item[POST /api/config/test] Test configuration feasibility
            \item[GET /api/config] Retrieve current configuration
            \item[PUT /api/config] Update and save configuration
        \end{description}
    \end{block}
    
    \begin{lstlisting}[language=Python, basicstyle=\ttfamily\tiny]
@app.route('/api/optimize', methods=['POST'])
def optimize():
    result = run_optimisation()
    if result['status'] == 'infeasible':
        analysis = result['analysis']
        return jsonify({"status": "infeasible", 
                       "issues": analysis['issues'],
                       "suggestions": analysis['suggestions']})
    return jsonify({"status": "success", 
                   "assignments": result['assignments']})
    \end{lstlisting}
\end{frame}
