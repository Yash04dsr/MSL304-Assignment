\section{Problem Statement \& Methodology}

\begin{frame}{Healthcare Staffing Challenge}
    \begin{block}{The Problem}
        Healthcare facilities face critical challenges in staff scheduling:
        \begin{itemize}
            \item \textbf{Patient demand} varies throughout the week
            \item \textbf{Staff availability} is constrained
            \item \textbf{Cost optimization} while maintaining service quality
            \item \textbf{Bottleneck identification} in patient flow
        \end{itemize}
    \end{block}
    
    \begin{exampleblock}{Solution Approach}
        Integrated web platform combining:
        \begin{itemize}
            \item Integer Linear Programming for staff optimization
            \item M/M/c Queue Theory for bottleneck analysis
            \item Modern web interface for real-time configuration
        \end{itemize}
    \end{exampleblock}
\end{frame}

\begin{frame}{Methodology Overview}
    \begin{block}{Dual-Module Approach}
        \begin{enumerate}
            \item \textbf{Patient Flow Simulation Module}
            \begin{itemize}
                \item Discrete event simulation using SimPy
                \item M/M/c queueing theory for bottleneck detection
                \item Real-time performance metrics
            \end{itemize}
            
            \item \textbf{Staff Optimization Module}
            \begin{itemize}
                \item Integer Linear Programming using PuLP
                \item Constraint satisfaction with CBC solver
                \item Infeasibility analysis and suggestions
            \end{itemize}
        \end{enumerate}
    \end{block}
    
    \begin{exampleblock}{Integration Strategy}
        REST API backend (Flask) + Modern web frontend (Bootstrap 5) + JSON configuration
    \end{exampleblock}
\end{frame}

\begin{frame}{Key Assumptions}
    \begin{alertblock}{Optimization Module}
        \begin{itemize}
            \item \textbf{Binary shifts}: Complete shift or none (no partials)
            \item \textbf{Uniform duration}: 8 hours per shift (AM/PM)
            \item \textbf{Fixed costs}: Constant cost per shift
            \item \textbf{Weekly horizon}: 5-day schedule (Mon-Fri)
            \item \textbf{Static availability}: Known in advance
        \end{itemize}
    \end{alertblock}
    
    \begin{block}{Simulation Module}
        \begin{itemize}
            \item \textbf{Markovian arrivals}: Poisson process ($\lambda$)
            \item \textbf{Exponential service}: Mean rate $\mu$
            \item \textbf{Homogeneous servers}: Identical service rates
            \item \textbf{FCFS discipline}: First-come, first-served
            \item \textbf{Infinite capacity}: No rejections
        \end{itemize}
    \end{block}
\end{frame}

\section{Mathematical Formulation}

\begin{frame}{Integer Linear Programming Model}
    \begin{block}{Sets \& Indices}
        $I$ : Staff members ($i \in I$) \quad | \quad $S$ : Shifts ($s \in S$)
    \end{block}
    
    \begin{block}{Parameters}
        \begin{itemize}
            \item $c_i$ : Cost per shift for staff $i$
            \item $R_s$ : Minimum required staff for shift $s$
            \item $H_i$ : Maximum weekly hours for staff $i$
            \item $A_{is}$ : Availability (1 if available, 0 otherwise)
            \item $h$ : Hours per shift (8 hours)
        \end{itemize}
    \end{block}
    
    \begin{block}{Decision Variables}
        \[
        x_{is} \in \{0,1\} \quad \forall i \in I, s \in S
        \]
        $x_{is} = 1$ if staff $i$ assigned to shift $s$
    \end{block}
\end{frame}

\begin{frame}{Optimization Model: Objective \& Constraints}
    \begin{block}{Objective Function}
        \[
        \text{Minimize: } Z = \sum_{i \in I} \sum_{s \in S} c_i \cdot x_{is}
        \]
        Minimize total staffing cost across all shifts
    \end{block}
    
    \begin{block}{Constraints}
        \begin{enumerate}
            \item \textbf{Shift Coverage:}
            \[
            \sum_{i \in I} x_{is} \geq R_s \quad \forall s \in S
            \]
            
            \item \textbf{Maximum Hours:}
            \[
            \sum_{s \in S} h \cdot x_{is} \leq H_i \quad \forall i \in I
            \]
            
            \item \textbf{Availability:}
            \[
            x_{is} \leq A_{is} \quad \forall i \in I, s \in S
            \]
            
            \item \textbf{Binary:}
            \[
            x_{is} \in \{0,1\} \quad \forall i \in I, s \in S
            \]
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}{Queue Theory: M/M/c Model}
    \begin{block}{System Parameters}
        \begin{itemize}
            \item $\lambda$ : Arrival rate (Poisson)
            \item $\mu$ : Service rate per server (Exponential)
            \item $c$ : Number of servers
            \item $\rho = \frac{\lambda}{\mu \cdot c}$ : Traffic intensity
        \end{itemize}
    \end{block}
    
    \begin{alertblock}{Stability Condition}
        $\rho = \frac{\lambda}{\mu \cdot c} < 1$ \quad (System stable when $\rho < 1$)
    \end{alertblock}
    
    \begin{block}{Performance Metrics}
        \begin{itemize}
            \item \textbf{Little's Law}: $L = \lambda \cdot W$
            \item $L_q$ : Average queue length
            \item $W_q$ : Average wait time
            \item $\rho$ : System utilization
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Bottleneck Detection Logic}
    \begin{block}{Traffic Intensity Classification}
        Based on $\rho = \frac{\lambda}{\mu \cdot c}$
    \end{block}
    
    \begin{alertblock}{Four-Level System}
        \begin{description}
            \item[\textcolor{red}{Critical}] $\rho \geq 0.95$ - System overload, immediate action
            
            \item[\textcolor{orange}{Warning}] $0.85 \leq \rho < 0.95$ - High congestion, add staff soon
            
            \item[\textcolor{yellow}{Caution}] $0.75 \leq \rho < 0.85$ - Moderate load, monitor
            
            \item[\textcolor{green}{Healthy}] $\rho < 0.75$ - Normal operation
        \end{description}
    \end{alertblock}
    
    \vspace{0.3cm}
    
    \begin{exampleblock}{Implementation}
        Queue length grows unbounded when critical | Wait times exceed limits
    \end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Infeasibility Analysis Engine}
    \begin{lstlisting}[language=Python, basicstyle=\ttfamily\tiny]
def analyze_infeasibility(config):
    issues = []
    suggestions = []
    
    # Check 1: Insufficient staff
    available_staff = len(config['staff'])
    max_requirement = max(config['shift_requirements'].values())
    if available_staff < max_requirement:
        issues.append("Insufficient staff available")
        suggestions.append(f"Hire {max_requirement - available_staff} more staff")
    
    # Check 2: No flexibility (staff can't cover shifts)
    for shift in config['shift_requirements']:
        available_for_shift = [s for s, data in config['staff'].items() 
                               if shift in data['availability']]
        if len(available_for_shift) < config['shift_requirements'][shift]:
            issues.append(f"Not enough staff available for {shift}")
    
    # ... more checks ...
    return {"issues": issues, "suggestions": suggestions}
    \end{lstlisting}
\end{frame}

        return jsonify({"status": "infeasible",
